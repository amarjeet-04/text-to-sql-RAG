### 4.
QUESTION: Show total bookings by supplier this month.
SQL:
SELECT 
    smr.SupplierName,
    COUNT(DISTINCT bd.PNRNo) AS total_bookings
FROM BookingData bd
INNER JOIN suppliermaster_Report smr 
    ON bd.SupplierId = smr.EmployeeId
WHERE 
    bd.CreatedDate >= '2026-02-01' 
    AND bd.CreatedDate < '2026-03-01'
    AND bd.BookingStatus NOT IN ('Cancelled', 'Not Confirmed', 'On Request')
GROUP BY smr.SupplierName

### 7.
QUESTION: Show bookings with check-in date in December 2023 by city.
SQL:
SELECT 
    b.PNRNo,
    b.DetailId,
    b.CheckInDate,
    b.CheckOutDate,
    b.AgentId,
    b.ProductName,
    mc.City,
    mc.Country
FROM BookingData b
INNER JOIN Master_City mc ON b.ProductCityId = mc.CityID
WHERE b.CheckInDate >= '2023-12-01' 
    AND b.CheckInDate < '2024-01-01'
    AND b.BookingStatus NOT IN ('Cancelled', 'Not Confirmed', 'On Request')
ORDER BY mc.City, b.CheckInDate;

### 10.
QUESTION: Which agent country generated the highest revenue in 2023?
SQL:
SELECT TOP 1
    am.AgentCountry,
    SUM(bd.AgentBuyingPrice) AS total_revenue
FROM BookingData bd
INNER JOIN AgentMaster_V1 am ON bd.AgentId = am.AgentId
WHERE bd.CreatedDate >= '2023-01-01'
    AND bd.CreatedDate < '2024-01-01'
    AND bd.BookingStatus NOT IN ('Cancelled', 'Not Confirmed', 'On Request')
    AND am.AgentCountry IS NOT NULL
GROUP BY am.AgentCountry
ORDER BY total_revenue DESC
